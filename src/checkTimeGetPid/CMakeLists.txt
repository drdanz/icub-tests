cmake_minimum_required(VERSION 3.0)
set(PROJECTNAME checkTimeGetPidsTest)
project(${PROJECTNAME})

include_directories(${CMAKE_SOURCE_DIR}
                    ${RTF_INCLUDE_DIRS}
                    ${YARP_INCLUDE_DIRS}
                    )

rtf_add_plugin(${PROJECT_NAME}    HEADERS checkTimeGetPid.h
                                  SOURCES checkTimeGetPid.cpp
                                  )

target_link_libraries(${PROJECTNAME} ${RTF_LIBRARIES}
                      ${YARP_LIBRARIES} YARP::YARP_rtf
                      YARP::YARP_manager)

install(TARGETS ${PROJECTNAME}
        EXPORT ${PROJECTNAME}
        COMPONENT runtime
        LIBRARY DESTINATION lib)
